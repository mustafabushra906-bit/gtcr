API Reference
Simple to integrate
Standard REST API with JSON. Works with any language.
REQUEST · Python
# Single baryon prediction
import requests

resp = requests.post(
    "https://api.gtcr.dev/predict",
    headers={"X-License-Key": "your-key"},
    json={
        "J": 1.5,   # spin
        "I": 0.0,   # isospin
        "S": 3,     # strangeness
        "n": 0      # radial excitation
    }
)
print(resp.json())
